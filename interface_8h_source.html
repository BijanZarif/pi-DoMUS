338 void
339 Interface<dim,spacedim,n_components,LAC>::
340 apply_neumann_bcs (
341  const typename DoFHandler<dim,spacedim>::active_cell_iterator &cell,
342  Scratch &scratch,
343  CopySystem &data,
344  Number &energy) const
345 {
346 
347  Number dummy = 0.0;
348 
349  for (unsigned int face=0; face < GeometryInfo<dim>::faces_per_cell; ++face)
350  {
351  unsigned int face_id = cell->face(face)->boundary_id();
352  if (cell->face(face)->at_boundary() && neumann_bcs.acts_on_id(face_id))
353  {
354  this->reinit(dummy, cell, face, scratch);
355 
356  auto &vars = scratch.get_values("solution", dummy);
357  auto &q_points = scratch.get_quadrature_points();
358  auto &JxW = scratch.get_JxW_values();
359 
360  for (unsigned int q=0; q<q_points.size(); ++q)
361  {
362  Vector<double> T(n_components);
363  neumann_bcs.get_mapped_function(face_id)->vector_value(q_points[q], T);
364 
365  for (unsigned int i=0; i < n_components; ++i)
366  if (neumann_bcs.get_mapped_mask(face_id)[i])
367  {
368  const std::vector<Number> &var_face = vars[q];
369 
370  energy -= (T[i]*var_face[i])*JxW[q];
371  }
372  }
373  break;
374  }
375  }
376 }

380 void
381 Interface<dim,spacedim,n_components,LAC>::
382 apply_forcing_terms (
383  const typename DoFHandler<dim,spacedim>::active_cell_iterator &cell,
384  Scratch &scratch,
385  CopySystem &data,
386  Number &energy) const
387 {
388  unsigned cell_id = cell->material_id();
389  if (forcing_terms.acts_on_id(cell_id))
390  {
391  Number dummy = 0.0;
392  this->reinit(dummy, cell, scratch);
393 
394  auto &vars = scratch.get_values("solution", dummy);
395  auto &q_points = scratch.get_quadrature_points();
396  auto &JxW = scratch.get_JxW_values();
397 
398  for (unsigned int q=0; q<q_points.size(); ++q)
399  {
400  const std::vector<Number> &var = vars[q]; // u,u,u
401  for (unsigned int i=0; i < n_components; ++i)
402  if (forcing_terms.get_mapped_mask(cell_id)[i])
403  {
404  double B = forcing_terms.get_mapped_function(cell_id)->value(q_points[q],i);
405  energy -= B*var[i]*JxW[q];
406  }
407  }
408  }
